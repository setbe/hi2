4. Two-pass module skeleton

namespace my {
struct plan { a::u32 count; };
struct input { a::u32 max_count; };
struct output { a::u32 written; };
}

namespace a::two_pass {
template<>
struct traits<my::plan, my::input> {
    static plan_result<my::plan> plan(const my::input& in) noexcept {
        if (in.max_count == 0) return {status::invalid_input, {}, {}};
        my::plan p{in.max_count};
        return {status::ok, p, {in.max_count * sizeof(a::u32), alignof(a::u32)}};
    }

    static build_result<my::output> build(const my::plan& p, a::arena_view arena) noexcept {
        auto r = arena.alloc(p.count * sizeof(a::u32), alignof(a::u32));
        if (!is_ok(r.code)) return {r.code, {0}};
        return {status::ok, {p.count}};
    }
};
}
